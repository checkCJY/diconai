## 함수별 상세

### `trigger_gas_alarms` — risk별 분기 (gas_alarm.py:75-161)

```python
def trigger_gas_alarms(gas_data, ingress_ts=None) -> list[dict]:
    sensor_id = gas_data.gas_sensor.id
    facility_id = gas_data.gas_sensor.facility_id
    cleared_gases: list[str] = []

    for gas in GAS_FIELDS:                                   # 9종
        risk = getattr(gas_data, f"{gas}_risk", None)
        value = getattr(gas_data, gas, None)
        if value is None: continue

        state_key = _state_key(sensor_id, gas)
        task_key = _task_key(sensor_id, gas)

        if risk == "danger":
            pending = cache.get(task_key)
            if pending: _revoke(pending); cache.delete(task_key)
            if gas in _AI_GUARDED_GASES and is_gas_ai_mute_active(...):
                continue                                     # ← AI 발화 직후 룰 60s 억제
            if try_transition(state_key, "danger", _CACHE_TTL):  # ← race-safe
                fire_danger_alarm_task.delay(sensor_id, gas, value, ...)

        elif risk == "warning":
            if get_state(state_key) in ("warning", "danger"): continue  # ← idempotent
            if gas in _AI_GUARDED_GASES and is_gas_ai_mute_active(...): continue
            if not cache.add(task_key, "_pending_", _TASK_KEY_TTL): continue  # SETNX
            task = fire_warning_alarm_task.apply_async(
                args=[...], countdown=WARNING_DURATION_SEC)   # 3초 후 발화
            cache.set(task_key, task.id, _TASK_KEY_TTL)
            try_transition(state_key, "warning", _CACHE_TTL)

        else:  # normal
            if cache.get(task_key):
                _revoke(cache.get(task_key)); cache.delete(task_key)
            if get_state(state_key) in ("warning", "danger"):
                cleared_gases.append(gas); clear_state(state_key)

    if cleared_gases:                                        # ← 9개 묶어 1번 발송
        fire_clear_notification_task.delay(sensor_id, source_label, cleared_gases)
    return []
```

**왜:**
- 9 가스 각각 독립 상태 머신 (`alarm:state:{sensor_id}:{gas}`). 다른 가스의 위험이 한 가스 알람을 막거나 트리거하지 않음.
- `try_transition` — atomic 천이 ("이전이 danger 아닐 때만 1회 fire"). Celery 다중 워커에서도 중복 발화 차단.
- `cache.add` (SETNX) — WARNING 첫 도착자만 타이머 시작. 1초당 1번 들어오는 데이터에서 같은 가스가 30초 연속 warning 일 때 30번 타이머 안 깔림.
- `_revoke` — risk 가 danger→normal 로 바뀌면 진행 중이던 warning 타이머 즉시 취소. "이미 위험 해제됐는데 3초 뒤 warning 띄우는" 시점 어긋남 방지.
- `_AI_GUARDED_GASES = {"co","h2s","co2"}` — fastapi 다변량 IF 추론 대상. AI 가 막 발화한 직후엔 룰 60s 억제 (option B, 중복 팝업 방지).
- `cleared_gases` 배치 — 9개 동시에 normal 복귀 시 9개 팝업 대신 1개 메시지로 묶음 (`[[operator_mental_model_simplicity]]`).

### `fire_danger_alarm_task` — DB 쓰기 + WS 푸시 분리 (tasks.py:140-221)

```python
@shared_task(bind=True, max_retries=3, default_retry_delay=5)
def fire_danger_alarm_task(self, sensor_id, gas_type, value, facility_id,
                            source_label, ingress_ts=None):
    # [H-5] DB 쓰기와 WS 푸시를 별도 try 블록으로 분리.
    # 이유: _push_to_ws 실패 시 retry 되면 create_alarm_and_event 가 재실행돼
    # AlarmRecord 가 중복 생성된다. DB 쓰기 성공 후 WS 푸시가 실패하면
    # retry 없이 경고 로그만 남긴다 — alarm 은 DB에 보존되므로 유실 없음.
    try:
        threshold = get_threshold_value(gas_type, "danger")
        summary = f"[긴급] {gas_name} 위험 수준 초과 ({value} {unit}) — 즉시 대피..."
        event, alarm = create_alarm_and_event(
            facility_id=..., alarm_type=AlarmType.GAS_THRESHOLD,
            sensor_id=sensor_id, gas_type=gas_type,
            measured_value=value, threshold_value=threshold,
            risk_level="danger", source_label=source_label, summary=summary,
            detected_at=timezone.now(),
        )
    except Exception as exc:
        raise self.retry(exc=exc)                            # ← DB 실패만 retry

    if event is not None and alarm is not None:
        ALARM_FIRED_TOTAL.labels(alarm_type="gas_threshold", risk_level="danger").inc()
        try:
            _push_to_ws({
                "event_id": event.id,
                "alarm_type": AlarmType.GAS_THRESHOLD,
                "gas_type": gas_type, "risk_level": "danger",
                "measured_value": value, "threshold_value": threshold,
                "source_label": source_label, "summary": summary,
                "message": alarm.get_short_message(),
                "event_ack_users": _get_event_ack_names(event.id),    # T3: 다중 관리자 시그널
                "is_new_event": alarm is not None,
                "ingress_ts": ingress_ts,
                "source": "static_legacy",                            # T4: 룰 단독 fire 표시
            })
        except Exception as exc:
            logger.warning("DANGER 알람 WS 푸시 실패 (DB 저장은 완료): %s", exc)
            return                                            # ← WS 실패는 retry X
```

**왜:**
- **2단 try 분리 (H-5)** — Celery retry 시 DB 쓰기가 재실행되면 AlarmRecord 중복. DB는 1회, WS만 실패 허용 (다음 alarm tick에서 자연 회복).
- `ingress_ts` 끝까지 전달 — E2E latency 측정에서 알람 push 시점까지 한 timeline 유지. 정량: POST→AlarmRecord ~1.7s ([../qa-7행-검증-2026-05-29.md](../qa-7행-검증-2026-05-29.md)).
- `event_ack_users` — T3 패치. 다중 관리자에서 운영자 A 가 "확인 중" 누른 알람이 재발화 시 토스트에 "(A 확인 중)" 시그널 (`[[alarm_t1_t3_pause_2026_05_20]]`).

### `_push_to_ws` — 5xx만 retry / 4xx는 그대로 (tasks.py:71-138)

```python
def _push_to_ws(alarm_data: dict, *, raise_on_failure: bool = True):
    alarm_data.setdefault("created_at", datetime.now(timezone.utc).isoformat())
    base = getattr(settings, "FASTAPI_INTERNAL_URL", "") or "http://127.0.0.1:8001"
    url = base.rstrip("/") + _ALARM_PUSH_PATH                # ← /internal/alarms/push/

    try:
        resp = httpx.post(url, json=alarm_data, headers=headers, timeout=10.0)
        if resp.status_code >= 500:                          # ← 5xx 만 retry 대상
            raise httpx.HTTPStatusError(...)
    except (httpx.RequestError, httpx.HTTPStatusError) as e:
        ALARM_WS_PUSH_FAILED_TOTAL.labels(alarm_type=...).inc()
        # ... silent → integration_log
    if not pushed and raise_on_failure:
        raise RuntimeError("FastAPI WS push failed")
```

**왜:**
- `timeout=10` — Phase 1 C5 보강. 기존 3초는 도커 네트워크 지연에 깨졌고, 실패 시 silent fail → 알람 영구 누락 사고가 있었음.
- 5xx vs 4xx 분리: 5xx는 일시 장애 (Redis 일시 점프) → retry 가치 있음. 4xx는 페이로드 검증 실패 → retry 의미 없음. retry 정책을 status 의미에 정확히 매핑.
- `raise_on_failure=False` 는 fire_clear_*_task 만 — 정상화 알림은 critical 아님, 손실 허용.

### `create_alarm_and_event` — Event 병합 + AlarmRecord (event_service.py:40-218)

핵심 분기 3가지:

```python
event_qs = Event.objects.select_for_update().filter(
    facility_id=..., event_type=alarm_type,
    status__in=[EventStatus.ACTIVE, ACKNOWLEDGED, IN_PROGRESS],
)
if sensor_id: event_qs = event_qs.filter(source_sensor_id=sensor_id)
active_event = event_qs.first()

if active_event:
    if not active_event.is_mergeable_time_window:    # 분기 ① 시간 초과 → 강제 RESOLVED
        active_event.status = EventStatus.RESOLVED
        active_event.save(update_fields=["status"])
        EventLog.objects.create(event=active_event, action=RESOLVED, note="최대 병합 시간 초과...")
        active_event = None       # 아래 새 Event 생성으로 falls through

    else:                                            # 분기 ② 병합 — Event 갱신만
        alarm = AlarmRecord.objects.create(event=active_event, ...)
        risk_escalated = RISK_LEVELS.index(risk_level) > RISK_LEVELS.index(active_event.risk_level)
        if risk_escalated:
            active_event.risk_level = risk_level
            EventLog.objects.create(action=STATUS_CHANGED, note=f"위험도 상승: {prev} → {risk_level}")

        cooldown = timedelta(seconds=settings.ALARM_REPOPUP_COOLDOWN_SEC)
        needs_renotify = (risk_escalated or
                          active_event.last_notified_at is None or
                          (timezone.now() - active_event.last_notified_at) >= cooldown)
        if needs_renotify:                            # ← 격상은 cooldown 무시
            active_event.last_notified_at = timezone.now()
            transaction.on_commit(lambda: _notify_safe(active_event))
            return active_event, alarm
        return active_event, None                     # cooldown 이내 — 재발송 X

if not active_event:                                 # 분기 ③ 신규
    policy = match_policy(event_type, facility_id, sensor_id, ...)
    event = Event.objects.create(..., policy=policy, ...)
    alarm = AlarmRecord.objects.create(event=event, ...)
    EventLog.objects.create(event=event, action=CREATED, new_status=ACTIVE)
    transaction.on_commit(lambda: _notify_safe(event))
    return event, alarm
```

**왜:**
- `select_for_update` — 같은 센서의 알람이 동시에 2개 워커에서 처리되면 Event 가 둘 생기는 race. 행 잠금으로 차단.
- **3분기**:
  - ① 같은 사건의 Event 이미 있고 시간 윈도우 초과 → 옛 Event 강제 종료 + 새 Event 생성 (사건 단위 분리)
  - ② 시간 윈도우 안 → AlarmRecord 만 추가, Event 는 갱신 (병합). 위험도 상승은 cooldown 무시.
  - ③ 활성 Event 없음 → 신규 Event + AlarmRecord + EventLog(CREATED) + AlertPolicy 매칭.
- `transaction.on_commit` — Notification 은 트랜잭션 커밋 후 호출. 롤백되는 Event 를 참조하는 race 차단.
- `risk_escalated`는 `EventLog`에 기록 (M-12) — WARNING→DANGER 천이가 사고 소급 분석에 중요.

## 핵심 dedup·race 방어 4종 정리

| 방어 | 메커니즘 | 막는 것 |
|---|---|---|
| `try_transition` | Redis cache 원자 비교-교환 | 다중 워커의 중복 발화 |
| `cache.add` (SETNX) | Redis SETNX | 같은 가스 1초 1번 × 30초 = 30 타이머 |
| `select_for_update` | PG row lock | 같은 사건의 Event 다중 생성 |
| `_push_to_ws` 후 retry 미수행 | tasks.py:213 return | WS 실패 후 AlarmRecord 중복 |
| Event cooldown (60s) | `last_notified_at` 비교 | 같은 Event 60초 안 재팝업 (단, 격상은 무시) |
| `_AI_GUARDED_GASES` mute | `is_gas_ai_mute_active` | AI 발화 직후 룰 중복 팝업 |
